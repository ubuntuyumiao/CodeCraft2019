//检查是否有车未到终点 
bool All_car_isreached(Car* car_array,int car_num_)
{
  bool all_car_isreached=true;
   for(int i=0;i<car_num_;i++)
    {
      //路上还有车等待调度
      if(car_array[i].state!=reached)  
      { 
	all_car_isreached=false;
	break;
      }

    }
return all_car_isreached;
}
